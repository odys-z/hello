Go Through Problem 2¶

Pseudo Code:

input H, R
B = 0
if H>48 then
    B = B + (H - 48) * 2 * R
    H = 48
end if
if H>40 then
   B = B + (H - 40) * (3/2) * R
   H = 40
end if
B = B + H * R
output B

def wdtpd_problem3(A, B):
    C = [0] * (len(A) * len(B))
    J, K, N = 0, 0, 0
    print('J K N Aj Bk Cn')
    print(J, K, N, A[J], B[K], C[N])
    while A[J] > 0:
        while B[K] < A[J]:
            C[N] = B[K]
            print(J, K, N, A[J], B[K], C[N])
            N = N + 1
            K = K + 1

        C[N] = A[J]
        print(J, K, N, A[J], B[K], C[N])
        N, J = N + 1, J + 1

    C[N] = B[K]
    return C

Q6, Intermediate Division¶

Psedo Code: see * handout: Programming Problem: Agram, 2017, Intermediate *

Python3 Validation:

def q6():
    a = [0] * 100
    for i in range( 2, 26 ):
        a[i] = i
    print(0, a)

    for k in range( 2, 26 ):
        if a[k] != 0:
            for j in range( 2*k, 26, k ):
                a[j] = 0
            print(k, a)

    s = 0
    for i in range(26):
        if a[i] > 0:
            s += 1
    print(s)
